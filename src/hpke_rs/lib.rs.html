234impl<Crypto: HpkeCrypto> Context<Crypto> {
235    /// 5.2. Encryption and Decryption
236    ///
237    /// Takes the associated data and the plain text as byte slices and returns
238    /// the ciphertext or an error.
239    ///
240    /// ```text
241    /// def Context.Seal(aad, pt):
242    ///   ct = Seal(self.key, self.ComputeNonce(self.seq), aad, pt)
243    ///   self.IncrementSeq()
244    ///   return ct
245    /// ```
246    pub fn seal(&mut self, aad: &[u8], plain_txt: &[u8]) -> Result<Ciphertext, HpkeError> {
247        let ctxt = Crypto::aead_seal(
248            self.hpke.aead_id,
249            &self.key,
250            &self.compute_nonce(),
251            aad,
252            plain_txt,
253        )?;
254        self.increment_seq()?;
255        Ok(ctxt)
256    }
257
258    /// 5.2. Encryption and Decryption
259    ///
260    /// Takes the associated data and the ciphertext as byte slices and returns
261    /// the plain text or an error.
262    ///
263    /// ```text
264    /// def Context.Open(aad, ct):
265    ///   pt = Open(self.key, self.ComputeNonce(self.seq), aad, ct)
266    ///   if pt == OpenError:
267    ///     raise OpenError
268    ///   self.IncrementSeq()
269    ///   return pt
270    /// ```
271    pub fn open(&mut self, aad: &[u8], cipher_txt: &[u8]) -> Result<Plaintext, HpkeError> {
272        let ptxt = Crypto::aead_open(
273            self.hpke.aead_id,
274            &self.key,
275            &self.compute_nonce(),
276            aad,
277            cipher_txt,
278        )?;
279        self.increment_seq()?;
280        Ok(ptxt)
281    }
282
283    /// 5.3. Secret Export
284    ///
285    /// Takes a serialised exporter context as byte slice and a length for the
286    /// output secret and returns an exporter secret as byte vector.
287    ///
288    /// ```text
289    /// def Context.Export(exporter_context, L):
290    ///  return LabeledExpand(self.exporter_secret, "sec", exporter_context, L)
291    ///```
292    pub fn export(&self, exporter_context: &[u8], length: usize) -> Result<Vec<u8>, HpkeError> {
293        labeled_expand::<Crypto>(
294            self.hpke.kdf_id,
295            &self.exporter_secret,
296            &self.hpke.ciphersuite(),
297            "sec",
298            exporter_context,
299            length,
300        )
301        .map_err(|e| HpkeError::CryptoError(format!("Crypto error: {}", e)))
302    }
303
304    /// def Context<ROLE>.ComputeNonce(seq):
305    ///     seq_bytes = I2OSP(seq, Nn)
306    ///     return xor(self.base_nonce, seq_bytes)
307    fn compute_nonce(&self) -> Vec<u8> {
308        let seq = self.sequence_number.to_be_bytes();
309        let mut enc_seq = vec![0u8; self.nonce.len() - seq.len()];
310        enc_seq.extend_from_slice(&seq);
311        util::xor_bytes(&enc_seq, &self.nonce)
312    }
313
314    /// def Context<ROLE>.IncrementSeq():
315    ///     if self.seq >= (1 << (8*Nn)) - 1:
316    ///       raise MessageLimitReached
317    ///     self.seq += 1
318    fn increment_seq(&mut self) -> Result<(), HpkeError> {
319        if u128::from(self.sequence_number)
320            >= ((1u128 << (8 * Crypto::aead_nonce_length(self.hpke.aead_id))) - 1)
321        {
322            return Err(HpkeError::MessageLimitReached);
323        }
324        self.sequence_number += 1;
325        Ok(())
326    }
327}

387    /// Set up an HPKE sender.
388    ///
389    /// For the base and PSK modes this encapsulates the public key `pk_r`
390    /// of the receiver.
391    /// For the Auth and AuthPSK modes this encapsulates and authenticates
392    /// the public key `pk_r` of the receiver with the senders secret key `sk_s`.
393    ///
394    /// **Note** that this API expects the public key to be encoded.
395    /// This differs from the RFC.
396    /// But the public keys will be present in encoded form rather than raw form
397    /// such that it doesn't make sense to deserialize before passing it in.
398    ///
399    /// The encapsulated secret is returned together with the context.
400    /// If the secret key is missing in an authenticated mode, an error is returned.
401    pub fn setup_sender(
402        &mut self,
403        pk_r: &HpkePublicKey,
404        info: &[u8],
405        psk: Option<&[u8]>,
406        psk_id: Option<&[u8]>,
407        sk_s: Option<&HpkePrivateKey>,
408    ) -> Result<(EncapsulatedSecret, Context<Crypto>), HpkeError> {
409        let randomness = self.random(self.kem_id.private_key_len())?;
410        let (zz, enc) = match self.mode {
411            Mode::Base | Mode::Psk => {
412                kem::encaps::<Crypto>(self.kem_id, pk_r.value.as_slice(), &randomness)?
413            }
414            Mode::Auth | Mode::AuthPsk => {
415                let sk_s = match sk_s {
416                    Some(s) => &s.value,
417                    None => return Err(HpkeError::InvalidInput),
418                };
419                kem::auth_encaps::<Crypto>(self.kem_id, pk_r.value.as_slice(), sk_s, &randomness)?
420            }
421        };
422        Ok((
423            enc,
424            self.clone().key_schedule(
425                &zz,
426                info,
427                psk.unwrap_or_default(),
428                psk_id.unwrap_or_default(),
429            )?,
430        ))
431    }

433    /// Set up an HPKE receiver.
434    ///
435    /// For the base and PSK modes this decapsulates `enc` with the secret key
436    /// `sk_r` of the receiver.
437    /// For the Auth and AuthPSK modes this decapsulates and authenticates `enc`
438    /// with the secret key `sk_r` of the receiver and the senders public key `pk_s`.
439    ///
440    /// **Note** that this API expects the public key to be encoded.
441    /// This differs from the RFC.
442    /// But the public keys will be present in encoded form rather than raw form
443    /// such that it doesn't make sense to deserialize before passing it in.
444    ///
445    /// The context based on the decapsulated values and, if present, the PSK is
446    /// returned.
447    /// If the secret key is missing in an authenticated mode, an error is returned.
448    pub fn setup_receiver(
449        &self,
450        enc: &[u8],
451        sk_r: &HpkePrivateKey,
452        info: &[u8],
453        psk: Option<&[u8]>,
454        psk_id: Option<&[u8]>,
455        pk_s: Option<&HpkePublicKey>,
456    ) -> Result<Context<Crypto>, HpkeError> {
457        let zz = match self.mode {
458            Mode::Base | Mode::Psk => kem::decaps::<Crypto>(self.kem_id, enc, &sk_r.value)?,
459            Mode::Auth | Mode::AuthPsk => {
460                let pk_s = match pk_s {
461                    Some(s) => s.value.as_slice(),
462                    None => return Err(HpkeError::InvalidInput),
463                };
464                kem::auth_decaps::<Crypto>(self.kem_id, enc, &sk_r.value, pk_s)?
465            }
466        };
467        self.clone().key_schedule(
468            &zz,
469            info,
470            psk.unwrap_or_default(),
471            psk_id.unwrap_or_default(),
472        )
473    }

586    /// Verify PSKs.
587    #[inline(always)]
588    fn verify_psk_inputs(&self, psk: &[u8], psk_id: &[u8]) -> Result<(), HpkeError> {
589        let got_psk = !psk.is_empty();
590        let got_psk_id = !psk_id.is_empty();
591        if (got_psk && !got_psk_id) || (!got_psk && got_psk_id) {
592            return Err(HpkeError::InconsistentPsk);
593        }
594
595        if got_psk && (self.mode == Mode::Base || self.mode == Mode::Auth) {
596            return Err(HpkeError::UnnecessaryPsk);
597        }
598        if !got_psk && (self.mode == Mode::Psk || self.mode == Mode::AuthPsk) {
599            return Err(HpkeError::MissingPsk);
600        }
601
602        // The PSK MUST have at least 32 bytes of entropy and SHOULD be of length Nh bytes or longer.
603        if (self.mode == Mode::Psk || self.mode == Mode::AuthPsk) && psk.len() < 32 {
604            return Err(HpkeError::InsecurePsk);
605        }
606
607        Ok(())
608    }

620    #[inline]
621    fn key_schedule_context(&self, info: &[u8], psk_id: &[u8], suite_id: &[u8]) -> Vec<u8> {
622        let psk_id_hash =
623            labeled_extract::<Crypto>(self.kdf_id, &[0], suite_id, "psk_id_hash", psk_id);
624        let info_hash = labeled_extract::<Crypto>(self.kdf_id, &[0], suite_id, "info_hash", info);
625        util::concat(&[&[self.mode as u8], &psk_id_hash, &info_hash])
626    }

628    /// Creating the Encryption Context
629    /// Generate the HPKE context from the given input.
630    pub fn key_schedule(
631        &self,
632        shared_secret: &[u8],
633        info: &[u8],
634        psk: &[u8],
635        psk_id: &[u8],
636    ) -> Result<Context<Crypto>, HpkeError> {
637        self.verify_psk_inputs(psk, psk_id)?;
638        let suite_id = self.ciphersuite();
639        let key_schedule_context = self.key_schedule_context(info, psk_id, &suite_id);
640        let secret =
641            labeled_extract::<Crypto>(self.kdf_id, shared_secret, &suite_id, "secret", psk);
642
643        let key = labeled_expand::<Crypto>(
644            self.kdf_id,
645            &secret,
646            &suite_id,
647            "key",
648            &key_schedule_context,
649            Crypto::aead_key_length(self.aead_id),
650        )
651        .map_err(|e| HpkeError::CryptoError(format!("Crypto error: {}", e)))?;
652        let base_nonce = labeled_expand::<Crypto>(
653            self.kdf_id,
654            &secret,
655            &suite_id,
656            "base_nonce",
657            &key_schedule_context,
658            Crypto::aead_nonce_length(self.aead_id),
659        )
660        .map_err(|e| HpkeError::CryptoError(format!("Crypto error: {}", e)))?;
661        let exporter_secret = labeled_expand::<Crypto>(
662            self.kdf_id,
663            &secret,
664            &suite_id,
665            "exp",
666            &key_schedule_context,
667            Crypto::kdf_digest_length(self.kdf_id),
668        )
669        .map_err(|e| HpkeError::CryptoError(format!("Crypto error: {}", e)))?;
670
671        Ok(Context {
672            key,
673            nonce: base_nonce,
674            exporter_secret,
675            sequence_number: 0,
676            hpke: self.clone(),
677        })
678    }
679
680    /// 4. Cryptographic Dependencies
681    /// Randomized algorithm to generate a key pair `(skX, pkX)` for the KEM.
682    /// This is equivalent to `derive_key_pair(random_vector(sk.len()))`
683    ///
684    /// Returns an `HpkeKeyPair`.
685    pub fn generate_key_pair(&mut self) -> Result<HpkeKeyPair, HpkeError> {
686        let (sk, pk) = kem::key_gen::<Crypto>(self.kem_id, &mut self.prng)?;
687        Ok(HpkeKeyPair::new(sk, pk))
688    }
689
690    /// 7.1.2. DeriveKeyPair
691    /// Derive a key pair for the used KEM with the given input key material.
692    ///
693    /// Returns an `HpkeKeyPair` result or an `HpkeError` if key derivation fails.
694    pub fn derive_key_pair(&self, ikm: &[u8]) -> Result<HpkeKeyPair, HpkeError> {
695        let (pk, sk) = kem::derive_key_pair::<Crypto>(self.kem_id, ikm)?;
696        Ok(HpkeKeyPair::new(sk, pk))
697    }
698
699    #[inline]
700    pub(crate) fn random(&mut self, len: usize) -> Result<Vec<u8>, HpkeError> {
701        let prng = &mut self.prng;
702        let mut out = vec![0u8; len];
703
704        #[cfg(feature = "hpke-test-prng")]
705        prng.try_fill_test_bytes(&mut out)
706            .map_err(|_| HpkeError::InsufficientRandomness)?;
707        #[cfg(not(feature = "hpke-test-prng"))]
708        prng.try_fill_bytes(&mut out)
709            .map_err(|_| HpkeError::InsufficientRandomness)?;
710
711        Ok(out)
712    }
713}

793        }
794
795        let mut different_bits = 0u8;
796        for (&byte_a, &byte_b) in self.value.iter().zip(other.value.iter()) {
797            different_bits |= byte_a ^ byte_b;
798        }
799        (1u8 & ((different_bits.wrapping_sub(1)).wrapping_shr(8)).wrapping_sub(1)) == 0
800    }
